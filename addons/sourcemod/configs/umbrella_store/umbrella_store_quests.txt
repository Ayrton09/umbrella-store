    //   reward_credits  = credits granted once on completion

    //   repeatable      = 1 allows the quest to be completed multiple times
    //   max_completions = max times a repeatable quest can reward (0 = unlimited)
    //   requires_quest  = optional quest id that must be completed first
    //   starts_at       = optional unix timestamp for activation start
    //   ends_at         = optional unix timestamp for activation end
