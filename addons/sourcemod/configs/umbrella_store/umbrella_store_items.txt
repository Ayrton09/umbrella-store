"Items"

    // =========================================================================
    // Umbrella Store item config
    // =========================================================================
    //
    // This file now ships with a full Source 2009 color preset already filled
    // in so server owners can test tags, name colors, and chat colors without
    // building a palette from scratch.
    //
    // Supported builtin item types:
    //   skin
    //   tag
    //   namecolor
    //   chatcolor
    //
    // Common supported keys:
    //   name
    //   type
    //   value
    //   model
    //   arms
    //   price
    //   team
    //   flag
    //   category
    //   description
    //   rarity
    //   sort_order
    //   icon
    //   sale_price
    //   sell_percent_override
    //   starts_at
    //   ends_at
    //   requires_item
    //   bundle_id
    //   hidden
    //   metadata
    //
    // Notes:
    //   - `value` must contain the real payload used by the item.
    //   - For tag items, `value` is the actual chat tag text and color tags.
    //   - For namecolor and chatcolor, `value` is just the color token.
    //   - For skin items, `model` is required and `arms` is optional.
    //   - The color preset below is aimed at Source 2009 games like CS:S.
    //   - CS:GO only guarantees the classic Multi-Colors subset.
    //
    // -------------------------------------------------------------------------
    // Example: basic tag
    // -------------------------------------------------------------------------
    // "tag_vip"
    // {
    //     "name" "VIP"
    //     "type" "tag"
    //     "category" "chat"
    //     "description" "Simple VIP chat tag."
    //     "value" "{orchid}[VIP]{default}"
    //     "price" "5000"
    //     "team" "0"
    // }
    //
    // -------------------------------------------------------------------------
    // Example: name color
    // -------------------------------------------------------------------------
    // "name_orchid"
    // {
    //     "name" "orchid"
    //     "type" "namecolor"
    //     "category" "chat"
    //     "description" "Applies orchid to the player name in chat."
    //     "value" "{orchid}"
    //     "price" "2500"
    //     "team" "0"
    // }
    //
    // -------------------------------------------------------------------------
    // Example: chat color
    // -------------------------------------------------------------------------
    // "chat_lightblue"
    // {
    //     "name" "lightblue"
    //     "type" "chatcolor"
    //     "category" "chat"
    //     "description" "Applies light blue to the player chat text."
    //     "value" "{lightblue}"
    //     "price" "2500"
    //     "team" "0"
    // }
    //
    // -------------------------------------------------------------------------
    // Example: player skin
    // -------------------------------------------------------------------------
    // "skin_ct_operator"
    // {
    //     "name" "CT Operator"
    //     "type" "skin"
    //     "category" "skins"
    //     "description" "Example Counter-Terrorist player skin."
    //     "model" "models/player/custom/example_ct/example_ct.mdl"
    //     "arms" "models/weapons/t_arms_example.mdl"
    //     "price" "25000"
    //     "sale_price" "12000"
    //     "team" "3"
    //     "rarity" "rare"
    //     "sort_order" "10"
    //     "icon" "icons/skins/ct_operator"
    //     "bundle_id" "starter_skins"
    //     "sell_percent_override" "55"
    // }
    //
    // -------------------------------------------------------------------------
    // Example: seasonal or timed item
    // -------------------------------------------------------------------------
    // "tag_halloween"
    // {
    //     "name" "Halloween"
    //     "type" "tag"
    //     "category" "events"
    //     "description" "Seasonal event tag."
    //     "value" "{orange}[HALLOWEEN]{default}"
    //     "price" "8000"
    //     "team" "0"
    //     "starts_at" "1790812800"
    //     "ends_at" "1791763199"
    //     "rarity" "event"
    // }
    //
    // -------------------------------------------------------------------------
    // Example: upgrade that requires another item first
    // -------------------------------------------------------------------------
    // "tag_vip_elite"
    // {
    //     "name" "VIP Elite"
    //     "type" "tag"
    //     "category" "chat"
    //     "description" "Upgrade tag that requires VIP first."
    //     "value" "{lightblue}[VIP+]{default}"
    //     "price" "12000"
    //     "team" "0"
    //     "requires_item" "tag_vip"
    // }
    //
    // -------------------------------------------------------------------------
    // Example: hidden/internal item
    // -------------------------------------------------------------------------
    // "tag_founder_hidden"
    // {
    //     "name" "Founder Hidden"
    //     "type" "tag"
    //     "category" "special"
    //     "description" "Hidden item example."
    //     "value" "{gold}[FOUNDER]{default}"
    //     "price" "0"
    //     "team" "0"
    //     "hidden" "1"
    //     "metadata" "grant=manual"
    // }
    //
    // -------------------------------------------------------------------------
    // Example: third-party custom type
    // -------------------------------------------------------------------------
    // A plugin must first register the type through the public API:
    //   US_RegisterItemType("player_badge", "cosmetics", true, false)
    //
    // "badge_founder"
    // {
    //     "name" "Founder Badge"
    //     "type" "player_badge"
    //     "category" "cosmetics"
    //     "description" "Example custom item type managed by another module."
    //     "value" "founder"
    //     "price" "15000"
    //     "team" "0"
    //     "metadata" "tier=founder;color=gold"
    // }
